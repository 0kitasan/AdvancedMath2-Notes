\begin{cases}
x=x(u,v) \\
y=y(u,v) \\ 
z=z(u,v) \\
\end{cases}
$$

$$
\begin{cases}
\vec{r_u}=(x_u,y_u,z_u) \\
\vec{r_v}=(x_v,y_v,z_v) \\ 
\end{cases}
$$

法向量
$$\vec{n}=\vec{r_u}\times \vec{r_v} $$
